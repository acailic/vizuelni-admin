## Task 5: Add Gallery Sort Functionality

**Files:**

- Modify: `src/components/demo-gallery/DemoGalleryClient.tsx`

- [ ] **Step 1: Add sort state and logic**

Add sort state after searchQuery state:

```tsx
const [searchQuery, setSearchQuery] = useState('');
const [sortBy, setSortBy] = useState<'newest' | 'alpha'>('newest');
```

Update filteredExamples to include sorting:

```tsx
const filteredExamples = useMemo(() => {
  let result =
    activeCategory === 'all'
      ? examples
      : examples.filter((example) => example.category === activeCategory);

  if (searchQuery.trim()) {
    const query = searchQuery.toLowerCase();
    result = result.filter((example) => {
      const title =
        example.title[locale === 'en' ? 'en' : 'sr']?.toLowerCase() || '';
      const tags = example.tags?.join(' ').toLowerCase() || '';
      const source = example.dataSource?.toLowerCase() || '';
      return (
        title.includes(query) || tags.includes(query) || source.includes(query)
      );
    });
  }

  // Sort results
  if (sortBy === 'alpha') {
    result = [...result].sort((a, b) => {
      const aTitle = a.title[locale === 'en' ? 'en' : 'sr'] || '';
      const bTitle = b.title[locale === 'en' ? 'en' : 'sr'] || '';
      return aTitle.localeCompare(bTitle);
    });
  } else if (sortBy === 'newest') {
    result = [...result].sort((a, b) => {
      const aDate = a.lastUpdated ? new Date(a.lastUpdated).getTime() : 0;
      const bDate = b.lastUpdated ? new Date(b.lastUpdated).getTime() : 0;
      return bDate - aDate;
    });
  }

  return result;
}, [activeCategory, examples, searchQuery, sortBy, locale]);
```

## Task 8: Add Data Catalog Visualizability Indicator

**Files:**

- Modify: `src/components/data/DataGovBrowser.tsx`

- [ ] **Step 1: Add visualizability check function**

Add after the pickAutoChartConfig function (around line 275):

```tsx
function checkVisualizability(resources: BrowseResource[]): {
  canVisualize: boolean;
  reason: string;
} {
  const previewableResources = resources.filter((r) =>
    isPreviewableFormat(r.format)
  );

  if (previewableResources.length === 0) {
    return {
      canVisualize: false,
      reason: locale === 'en' ? 'No CSV/JSON' : 'Нема CSV/JSON',
    };
  }

  return {
    canVisualize: true,
    reason:
      locale === 'en'
        ? `${previewableResources.length} chartable`
        : `${previewableResources.length} за график`,
  };
}
```

- [ ] **Step 2: Add visualizability badge to dataset cards**
